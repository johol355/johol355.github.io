2 Define a data-generating process

See the DAG below for a quick overview of the DGP. The idea is that our binary treatment, X, has a causal effect on the binary outcome Y. There is a set of confounders, C_{i}. C_{1} is binary and has a linear effect (on the log-odds scale) on Y. C_{2} is categorical (1-365), as is C_{3} (1-24). C_{2} and C_{3} both affect the probability of treatment and outcome in a non-linear fashion (by way of cosine functions). See the code (function dataset) for details. The function allows for differences between subgroups of individuals (eg. belonging to different centers) w.r.t. baseline risk, treatment effect and mechanism of confounding.

def dgp(n_samples, treatment_effect = 0.5, non_linear_effect = 1):
    c1 = rng.choice(a=2, size=n_samples)
    c2 = rng.choice(a=365, size=n_samples)
    c3 = rng.choice(a=24, size=n_samples)
    center = rng.choice(a=['a', 'b', 'c'], size=n_samples)

    center_baseline_outcome_risk = {
        'a': -1.5,
        'b': -1.0,
        'c': -2.0
    }
    
    center_treatment_modifier = {
        'a': 1.0,
        'b': 0,
        'c': 2.0
    }

    def f_x(c2, c3):
        return non_linear_effect * (np.cos(np.pi/(365/2)*c2) + np.cos(np.pi/(24/2)*c3))
    def f_y(c2, c3):
        return non_linear_effect * (np.cos(np.pi/(365/2)*c2) + np.cos(np.pi/(24/2)*c3))

    p_x = expit(0 + 1.5 * c1 + 2 * f_x(c2, c3))
    x = [rng.choice(a=2, p=[1-p, p]) for p in p_x]
    p_y = expit([center_baseline_outcome_risk[c] for c in center] + c1 + f_y(c2, c3) + np.array([center_treatment_modifier[c] for c in center]) * treatment_effect * np.array(x))
    p_y0 = expit([center_baseline_outcome_risk[c] for c in center] + c1 + f_y(c2, c3) + np.array([center_treatment_modifier[c] for c in center]) * treatment_effect * np.zeros(n_samples))
    p_y1 = expit([center_baseline_outcome_risk[c] for c in center] + c1 + f_y(c2, c3) + np.array([center_treatment_modifier[c] for c in center]) * treatment_effect * np.ones(n_samples))
    ITE = p_y1 - p_y0

    y = [rng.choice(a=2, p=[1-p, p]) for p in p_y]
    d = {'C1': c1,
         'C2': c2,
         'C3': c3,
         'CENTER': center,
         'X': x,
         'Y': y,
         'ITE': ITE}
    return pd.DataFrame(d)

4 Model comparison

Four models will be compared: * A naive (unadjusted model) * Logistic regression with X, C_{1}, C_{2} and C_{3} as predictors * Full Luxury Bayesian Inference I: A Bayesian Network with linear dependencies between parameters (i.e. an elitist logistic regression) * Full Luxury Bayesian Inference II: A Bayesian Network with Gaussian processes (Hilbert Space approximations)

4.1 Naive model, definition

y_{i} \sim Bernoulli(p_{i})

p_{i} = invlogit(\alpha + \beta_{XY[idx_{X}]})

\alpha \sim N(0, 1)

\left( \begin{array}{c} \beta_{XY[idx_{X=0}]} \sim N(0, 1) \\ \beta_{XY[idx_{X=1}]} \sim N(0, 1) \end{array} \right)

x \in \{0, 1\}

COORDS = {'sample': np.arange(0, n_samples), 'confound': [0,1], 'treatment': [0,1], 'outcome': [0,1], 'day': np.arange(365), 'hour': np.arange(24)}
with pm.Model(coords=COORDS) as naive_m:
    C1 = pm.Bernoulli('C1', p=.5, dims='sample')
    C2 = pm.Categorical('C2', p=np.ones(365)/365, dims='sample')
    C3 = pm.Categorical('C3', p=np.ones(24)/24, dims='sample')

    X = pm.Bernoulli('X', p=.5, dims='sample')

    # outcome model
    alpha_Y = pm.Normal('alpha_Y', 0, 1)
    beta_XY = pm.Normal('beta_XY', 0, 1, dims='treatment')
    p_Y = pm.Deterministic('p_Y', pm.math.invlogit(alpha_Y + beta_XY[X]), dims='sample')

    Y = pm.Bernoulli('Y', p_Y, dims='sample')

4.3.0.1 Outcome part

y_{i} \sim Bernoulli(p_{yi})

p_{yi} = invlogit(\alpha_{y} + \beta_{XY[idx_{X=x}]} + \beta_{C_{1}Y[idx_{C_{1}}=c_{1}]} + f_{C_{2}Y}(c_{2}) + f_{C_{3}Y}(c_{3}))

\left( \begin{array}{c} \beta_{XY[idx_{X=0}]} \sim N(0, 2) \\ \beta_{XY[idx_{X=1}]} \sim N(0, 2) \end{array} \right)

\left( \begin{array}{c} \beta_{C_{1}Y[idx_{C_{1}=0}]} \sim N(0, 2) \\ \beta_{C_{1}Y[idx_{C_{1}=1}]} \sim N(0, 2) \end{array} \right)

\theta_{cy_{3}} = \ell_2, \text{period}_{2}

\ell_2 \sim InvGamma(2, 1)

\text{period}_{2} = 24

    C1 = pm.Bernoulli('C1', p=.5, dims='sample')
    C2 = pm.Categorical('C2', p=np.ones(365)/365, dims='sample')
    C3 = pm.Categorical('C3', p=np.ones(24)/24, dims='sample')

    # trt model
    alpha_X = pm.Normal('alpha_X', 0, 2)
    beta_CX = pm.Normal('beta_CX', 0, 2, dims='confound')

    ls_trt_C2 = pm.Gamma('ls_trt_C2', 4, .25)
    eta_trt_C2 = pm.HalfNormal('eta_trt_C2', 1)
    cov_trt_C2 = pm.gp.cov.Periodic(1, ls=ls_trt_C2, period=365)
    gp_trt_C2 = pm.gp.HSGPPeriodic(m=20, cov_func=cov_trt_C2, scale=eta_trt_C2**2)
    f_trt_C2 = gp_trt_C2.prior('f_trt_C2', X=C2[:,None])

    ls_trt_C3 = pm.Gamma('ls_trt_C3', 2, 1)
    eta_trt_C3 = pm.HalfNormal('eta_trt_C3', 1)
    cov_trt_C3 = pm.gp.cov.Periodic(1, ls=ls_trt_C3, period=24)
    gp_trt_C3 = pm.gp.HSGPPeriodic(m=20, cov_func=cov_trt_C3, scale=eta_trt_C3**2)
    f_trt_C3 = gp_trt_C3.prior('f_trt_C3', X=C3[:,None])

    p_X = pm.Deterministic('p_X', pm.math.invlogit(alpha_X + f_trt_C2 + f_trt_C3), dims='sample')
    X = pm.Bernoulli('X', p=p_X, dims='sample')

    # outcome model
    alpha_Y = pm.Normal('alpha_Y', -1, 2)
    beta_CY = pm.Normal('beta_CY', 0, 2, dims='confound')
    beta_XY = pm.Normal('beta_XY', 0, 2, dims='treatment')

    ls_out_C2 = pm.Gamma('ls_out_C2', 4, .25)
    eta_out_C2 = pm.HalfNormal('eta_out_C2', 1)
    cov_out_C2 = pm.gp.cov.Periodic(1, ls=ls_out_C2, period=365)
    gp_out_C2 = pm.gp.HSGPPeriodic(m=20, cov_func=cov_out_C2, scale=eta_out_C2**2)
    f_out_C2 = gp_out_C2.prior('f_out_C2', X=C2[:,None])

    ls_out_C3 = pm.Gamma('ls_out_C3', 2, 2)
    eta_out_C3 = pm.HalfNormal('eta_out_C3', 1)
    cov_out_C3 = pm.gp.cov.Periodic(1, ls=ls_out_C3, period=25)

    f_out_C3 = gp_out_C3.prior('f_out_C3', X=C3[:,None])


    p_Y = pm.Deterministic('p_Y', pm.math.invlogit(alpha_Y + beta_CY[C1] + beta_XY[X] + f_out_C2 + f_out_C3), dims='sample')

    Y = pm.Bernoulli('Y', p_Y, dims='sample')

def train_model(model, observations={'X': dataset.X, 'Y': dataset.Y, 'C1': dataset.C1, 'C2': dataset.C2, 'C3': dataset.C3}):
    model_trained = observe(model, observations)
    with model_trained:
      #  advi = pm.fit(method='ADVI', callbacks=[CheckParametersConvergence()])
      #  model_trained_idata = advi.sample(5000)
        model_trained_idata = pm.sample()
    return model_trained, model_trained_idata

def create_counterfactual_models(model, observations={'C1': dataset.C1, 'C2': dataset.C2, 'C3': dataset.C3}):
        model_counterfactual = do(model, observations)
        model_hems0 = do(model_counterfactual, {'X': np.zeros(n_samples, dtype='int32')})
        model_hems1 = do(model_counterfactual, {'X': np.ones(n_samples, dtype='int32')})
        return model_hems0, model_hems1

def sample_counterfactual_model(model_do_0, model_do_1, training_idata, SEED=19):
    model_hems0_idata = pm.sample_posterior_predictive(
        training_idata,
        model_do_0,
        var_names=['p_Y', 'Y'],
        predictions=True,
        random_seed=SEED,
        )

    model_hems1_idata = pm.sample_posterior_predictive(
        training_idata,
        model_do_1,
        var_names=['p_Y', 'Y'],
        predictions=True,
        random_seed=SEED)
    
    return model_hems0_idata, model_hems1_idata

def calculate_ate_contrast(do0_idata, do1_idata):
    ate = (do1_idata.predictions.p_Y - do0_idata.predictions.p_Y).mean(dim='sample')
    return ate

def run_counterfactuals(models):
    model_names = []
    model_ates = []
    model_idata_dict = {}

    for key in models.keys():
        model = models[key]
        model_trained, model_idata = train_model(model)
        model_idata_dict[key] = model_idata
        model_hems0, model_hems1 = create_counterfactual_models(model_trained)
        model_hems0_idata, model_hems1_idata = sample_counterfactual_model(model_hems0, model_hems1, model_idata)
        model_ate = calculate_ate_contrast(model_hems0_idata, model_hems1_idata)

        model_names.append(key)
        model_ates.append(model_ate)
    return model_names, model_ates, model_idata_dict
